#define BUTTON_PIN 0
#define BUZZER_LED 15
#define RST_PIN 4
#define SS_PIN 5
#define HALL_SENSOR_PIN 34

Stepper drum(2048, 13, 12, 14, 27);
ThreeWire myWire(16, 17, 2);  
RtcDS1302ThreeWire Rtc(myWire);
Adafruit_SSD1306 display(128, 64, &Wire, -1);
MFRC522 rfid(SS_PIN, RST_PIN);
Servo loadServo;

unsigned long buttonPressStart = 0;
bool rfidAuthenticated = false;
bool isLidOpen = false;
bool systemStarted = false;

void showMsg(String l1, String l2);
void showMsg(String l1, String l2, String l3);
void syncTimeByNTP();
void displayClock();
void checkRFID();
void handleButtonLogic();
void toggleLoadingMode();
void triggerBuzzer();
void startSystem();

 ===================== SETUP =====================
void setup() {
  Serial.begin(115200);

  pinMode(BUTTON_PIN, INPUT_PULLUP);
  pinMode(BUZZER_LED, OUTPUT);
  pinMode(HALL_SENSOR_PIN, INPUT);

  Wire.begin();
  delay(200);

  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  delay(200);

  display.ssd1306_command(SSD1306_DISPLAYON);
  display.clearDisplay();
  display.display();
  delay(200);
}

 ===================== LOOP =====================
void loop() {
  if (!systemStarted) {
    if (digitalRead(BUTTON_PIN) == LOW) {
      delay(300);

      display.ssd1306_command(SSD1306_DISPLAYON);
      startSystem();
      systemStarted = true;
    }
    return;
  }

  if (!rfidAuthenticated) {
    displayClock();
    checkRFID();
  } else {
    handleButtonLogic();
  }
}

 ===================== START SYSTEM =====================
void startSystem() {

  showMsg(STARTING, Motor calibration...);

  loadServo.attach(25);
  loadServo.write(90);
  drum.setSpeed(10);

   Homing
  while (digitalRead(HALL_SENSOR_PIN) == HIGH) {
    drum.step(1);
    delay(2);
  }

  Rtc.Begin();

  WiFiManager wm;
  showMsg(WIFI SETUP, Connect to, StellHeal_Setup);

  if (!wm.autoConnect(StellHeal_Setup)) {
    ESP.restart();
  }

  syncTimeByNTP();

  SPI.begin();
  rfid.PCD_Init();

  showMsg(System Ready, Ready);
  delay(2000);
}

 ===================== NTP =====================
void syncTimeByNTP() {
  showMsg(TIME SYNC, Getting time...);

  setenv(TZ, EET-2EEST,M3.5.03,M10.5.04, 1);
  tzset();

  configTime(0, 0, pool.ntp.org, time.nist.gov);

  struct tm timeinfo;
  int retry = 0;

  while (!getLocalTime(&timeinfo) && retry  10) {
    delay(500);
    retry++;
    showMsg(WAIT NTP, String(retry));
  }

  if (retry  10) {
    showMsg(TIME OK, Saved);

    time_t now;
    time(&now);

    RtcDateTime rtcTime;
    rtcTime.InitWithUnix32Time(now);
    Rtc.SetDateTime(rtcTime);

  } else {
    showMsg(NTP FAILED, RTC only);
  }

  delay(1000);
}

 ===================== CLOCK (RTC ONLY) =====================
void displayClock() {
  static unsigned long lastUpdate = 0;
  if (millis() - lastUpdate  1000) return;
  lastUpdate = millis();

  RtcDateTime now = Rtc.GetDateTime();

  display.clearDisplay();
  display.setCursor(20, 20);
  display.setTextSize(3);
  display.setTextColor(WHITE);

  display.printf(%02d%02d, now.Hour(), now.Minute());
  display.display();
}

 ===================== RFID =====================
void checkRFID() {
  if (rfid.PICC_IsNewCardPresent() && rfid.PICC_ReadCardSerial()) {
    rfidAuthenticated = true;
    triggerBuzzer();
    showMsg(AUTH OK, Button Active);
    delay(1000);
  }
}

 ===================== BUTTON =====================
void handleButtonLogic() {
  static bool buzzerTriggered = false;

  if (digitalRead(BUTTON_PIN) == LOW) {
    if (buttonPressStart == 0) buttonPressStart = millis();

    unsigned long duration = millis() - buttonPressStart;

    if (duration = 10000) {
      if (!buzzerTriggered) { triggerBuzzer(); buzzerTriggered = true; }
      showMsg(ACTION, RELEASE OFF);
    }
    else if (duration = 5000) {
      if (!buzzerTriggered) { triggerBuzzer(); buzzerTriggered = true; }
      showMsg(ACTION, RESET WIFI);
    }
  } else {
    if (buttonPressStart  0) {
      unsigned long finalDuration = millis() - buttonPressStart;
      buttonPressStart = 0;
      buzzerTriggered = false;

      if (finalDuration = 10000) {
        showMsg(SHUTDOWN, Sleeping...);
        delay(2000);

        display.clearDisplay();
        display.display();
        display.ssd1306_command(SSD1306_DISPLAYOFF);
        delay(100);

        esp_sleep_enable_ext0_wakeup(GPIO_NUM_0, 0);
        esp_deep_sleep_start();
      }
      else if (finalDuration = 5000) {
        showMsg(RESET, WiFi...);
        WiFiManager wm;
        wm.resetSettings();
        ESP.restart();
      }
      else if (finalDuration  50 && finalDuration  3000) {
        toggleLoadingMode();
      }
    }
  }
}

 ===================== SERVO =====================
void toggleLoadingMode() {
  if (!isLidOpen) {
    showMsg(FILLING, Add meds);
    loadServo.write(0);
    isLidOpen = true;
  } else {
    showMsg(CLOSING, Exit...);
    loadServo.write(90);
    isLidOpen = false;
    delay(1000);
    rfidAuthenticated = false;
  }
}

 ===================== BUZZER =====================
void triggerBuzzer() {
  digitalWrite(BUZZER_LED, HIGH);
  delay(200);
  digitalWrite(BUZZER_LED, LOW);
}
